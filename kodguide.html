Databas — Prisma schema

Alla tabeller definieras här. Kör npx prisma db push för att skapa dem i databasen.

backend/prisma/schema.prisma
generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

model Project {
  id          Int        @id @default(autoincrement())
  name        String
  description String?
  status      String     @default("GREEN")   // GREEN, YELLOW, RED
  phase       String?
  budget      Float?
  summary     String?
  createdAt   DateTime   @default(now())
  updatedAt   DateTime   @updatedAt

  activities  Activity[]
  risks       Risk[]
  issues      Issue[]
  decisions   Decision[]
  shareTokens ShareToken[]
}

model Activity {
  id          Int      @id @default(autoincrement())
  project     Project  @relation(fields: [projectId], references: [id], onDelete: Cascade)
  projectId   Int
  name        String
  startDate   DateTime
  endDate     DateTime
  progress    Int      @default(0)            // 0–100
  status      String   @default("NOT_STARTED") // NOT_STARTED, IN_PROGRESS, DONE, BLOCKED
  isMilestone Boolean  @default(false)
  budget      Float?
  notes       String?
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  resources  ActivityResource[]
  dependsOn  Dependency[]  @relation("DependencyTo")
  blockedBy  Dependency[]  @relation("DependencyFrom")
}

model Dependency {
  id             Int      @id @default(autoincrement())
  fromActivity   Activity @relation("DependencyFrom", fields: [fromActivityId], references: [id], onDelete: Cascade)
  fromActivityId Int
  toActivity     Activity @relation("DependencyTo", fields: [toActivityId], references: [id], onDelete: Cascade)
  toActivityId   Int
}

model Domain {
  id     Int     @id @default(autoincrement())
  name   String
  tribes Tribe[]
}

model Tribe {
  id       Int    @id @default(autoincrement())
  name     String
  domain   Domain @relation(fields: [domainId], references: [id])
  domainId Int
  teams    Team[]
}

model Team {
  id        Int        @id @default(autoincrement())
  name      String
  tribe     Tribe      @relation(fields: [tribeId], references: [id])
  tribeId   Int
  resources Resource[]
}

model Resource {
  id         Int               @id @default(autoincrement())
  name       String
  role       String?
  team       Team              @relation(fields: [teamId], references: [id])
  teamId     Int
  activities ActivityResource[]
}

model ActivityResource {
  activity    Activity @relation(fields: [activityId], references: [id], onDelete: Cascade)
  activityId  Int
  resource    Resource @relation(fields: [resourceId], references: [id])
  resourceId  Int
  hoursNeeded Float?
  @@id([activityId, resourceId])
}

model Risk {
  id          Int      @id @default(autoincrement())
  project     Project  @relation(fields: [projectId], references: [id], onDelete: Cascade)
  projectId   Int
  description String
  impact      String   @default("MEDIUM")  // LOW, MEDIUM, HIGH
  probability String   @default("MEDIUM")
  status      String   @default("OPEN")    // OPEN, MITIGATED, CLOSED
  owner       String?
  createdAt   DateTime @default(now())
}

model Issue {
  id          Int      @id @default(autoincrement())
  project     Project  @relation(fields: [projectId], references: [id], onDelete: Cascade)
  projectId   Int
  description String
  status      String   @default("OPEN")  // OPEN, IN_PROGRESS, RESOLVED
  owner       String?
  createdAt   DateTime @default(now())
}

model Decision {
  id          Int      @id @default(autoincrement())
  project     Project  @relation(fields: [projectId], references: [id], onDelete: Cascade)
  projectId   Int
  description String
  owner       String?
  decidedAt   DateTime @default(now())
}

model ShareToken {
  id        Int       @id @default(autoincrement())
  project   Project   @relation(fields: [projectId], references: [id], onDelete: Cascade)
  projectId Int
  token     String    @unique
  expiresAt DateTime?
  createdAt DateTime  @default(now())
}

import express from 'express'
import cors from 'cors'
import swaggerJsdoc from 'swagger-jsdoc'
import swaggerUi from 'swagger-ui-express'

import projectRoutes from './routes/projects'
import activityRoutes from './routes/activities'
import resourceRoutes from './routes/resources'
import raidRoutes from './routes/raid'
import shareRoutes from './routes/share'

const app = express()
const PORT = process.env.PORT || 4000

app.use(cors())
app.use(express.json())

// Swagger — dokumentation genereras automatiskt från JSDoc-kommentarer i routes
const swaggerOptions: swaggerJsdoc.Options = {
  definition: {
    openapi: '3.0.0',
    info: { title: 'Projekt X API', version: '1.0.0' },
    servers: [{ url: `http://localhost:${PORT}` }],
  },
  apis: ['./src/routes/*.ts'],
}
app.use('/api/docs', swaggerUi.serve, swaggerUi.setup(swaggerJsdoc(swaggerOptions)))

// Routes
app.use('/api/projects',   projectRoutes)
app.use('/api/activities', activityRoutes)
app.use('/api/resources',  resourceRoutes)
app.use('/api/raid',       raidRoutes)
app.use('/api/share',      shareRoutes)

app.get('/api/health', (_req, res) => res.json({ status: 'ok' }))

app.listen(PORT, () => {
  console.log(`Backend running on http://localhost:${PORT}`)
  console.log(`Swagger docs: http://localhost:${PORT}/api/docs`)
})
Mönstret för en route-fil — alla routes följer samma struktur. Skapa en ny fil i src/routes/ och registrera den i index.ts.
backend/src/routes/[namn].ts — mönster
import { Router, Request, Response } from 'express'
import prisma from '../lib/prisma'

const router = Router()

// GET — hämta alla
router.get('/', async (req: Request, res: Response) => {
  const items = await prisma.project.findMany()
  res.json(items)
})

// GET — hämta en
router.get('/:id', async (req: Request, res: Response) => {
  const item = await prisma.project.findUnique({ where: { id: Number(req.params.id) } })
  if (!item) return res.status(404).json({ error: 'Hittades inte' })
  res.json(item)
})

// POST — skapa
router.post('/', async (req: Request, res: Response) => {
  const item = await prisma.project.create({ data: req.body })
  res.status(201).json(item)
})

// PUT — uppdatera
router.put('/:id', async (req: Request, res: Response) => {
  const item = await prisma.project.update({ where: { id: Number(req.params.id) }, data: req.body })
  res.json(item)
})

// DELETE — ta bort
router.delete('/:id', async (req: Request, res: Response) => {
  await prisma.project.delete({ where: { id: Number(req.params.id) } })
  res.status(204).send()
})

export default router

frontend/src/pages/[Sida].tsx — mönster
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query'
import { getProjects, createProject } from '../lib/api'

export default function MinSida() {
  const qc = useQueryClient()

  // Hämta data
  const { data: projects = [] } = useQuery({
    queryKey: ['projects'],
    queryFn: () => getProjects(),
  })

  // Skapa/uppdatera data
  const mutation = useMutation({
    mutationFn: (data: object) => createProject(data),
    onSuccess: () => qc.invalidateQueries({ queryKey: ['projects'] }),
  })

  return (
    <div className="p-6">
      <h1 className="text-xl font-bold mb-4">Min sida</h1>
      {projects.map((p: any) => (
        <div key={p.id}>{p.name}</div>
      ))}
    </div>
  )
}
